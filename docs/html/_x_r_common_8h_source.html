70  #define XR_CALL( xrCall, xrLogger, bThrow ) CheckXrResult( xrCall, xrLogger, bThrow, #xrCall, __FILE__, __LINE__ );
71  inline XrResult CheckXrResult(
72  XrResult xrResult,
73  std::shared_ptr< spdlog::logger > xrLogger,
74  bool bThrow,
75  const char *xrCall = nullptr,
76  const char *srcFile = nullptr,
77  int srcLineNum = 0 )
78  {
79  if ( XR_UNQUALIFIED_SUCCESS( xrCall ) )
80  {
81  const char *xrEnumStr = XrEnumToString( xrResult );
82  std::string eMessage = "Error on OpenXR call ";
83  if ( xrCall )
84  eMessage.append( xrCall );
85 
86  xrLogger->error( "{}. Error ({}) {} in file {} line {}", eMessage, std::to_string( xrResult ), xrEnumStr, srcFile, srcLineNum );
87 
88  if ( bThrow )
89  throw std::runtime_error( eMessage );
90  }
91  else
92  {
93  xrLogger->info( "Success: {}", xrCall );
94  }
95 
96  return xrResult;
97  }
