61  #define XR_CALL( xrCall, xrLogger, bThrow ) CheckXrResult( xrCall, xrLogger, bThrow, #xrCall, __FILE__, __LINE__ );
62  inline XrResult CheckXrResult(
63  XrResult xrResult,
64  std::shared_ptr< spdlog::logger > xrLogger,
65  bool bThrow,
66  const char *xrCall = nullptr,
67  const char *srcFile = nullptr,
68  int srcLineNum = 0 )
69  {
70  if ( XR_UNQUALIFIED_SUCCESS( xrCall ) )
71  {
72  const char *xrEnumStr = ENUM_STR( xrResult );
73  std::string eMessage = "Error on OpenXR call ";
74  if ( xrCall )
75  eMessage.append( xrCall );
76 
77  xrLogger->error( "{}. Error ({}) {} in file {} line {}", eMessage, std::to_string( xrResult ), xrEnumStr, srcFile, srcLineNum );
78 
79  if ( bThrow )
80  throw std::runtime_error( eMessage );
81  }
82  else
83  {
84  xrLogger->info( "Success: {}", xrCall );
85  }
86 
87  return xrResult;
88  }
